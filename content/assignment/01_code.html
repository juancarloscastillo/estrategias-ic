## Calculate age in years, months, or days
calc_age <- function(birth_date, ref_date = Sys.Date(), unit = "year") {

    require(lubridate)

    period <- as.period(interval(birth_date, ref_date),
                        unit = unit)

    switch(unit,
           year = year(period),
           month = month(period),
           day = day(period),
           stop = "Unknown time unit. Choose year, month, or day.")

}
